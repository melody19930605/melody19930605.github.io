利用Python生成C#实体

2019-12-10
# -*- coding: UTF-8 -*-

import pymysql


def TypeChange(typeName):
    if typeName == "bigint":
        return "long"
    elif typeName == "varchar":
        return "string"
    elif typeName == "datetime":
        return "DateTime"
    elif typeName == "int":
        return "int"
    elif typeName == "decimal":
        return "decimal"
    else:
        return ""


def ColumnNameChange(columnName, prefix):
    if len(columnName) < len(prefix):
        return columnName
    elif columnName[0:len(prefix)] == prefix:
        return columnName[len(prefix):len(columnName)]
    else:
        return columnName


tableName = input("请输入表名：")
dataModelName = input("请输入实体名称：")
prefix = input("字段前缀名：")

# 新建数据库连接
conn = pymysql.connect(
    host='',
    port=,
    user='',
    password='',
    db='',
    charset='utf8')
# 2获取游标
cursor = conn.cursor()
# 3执行查询，并获取查询的总行数
rowNums = cursor.execute('''SELECT COLUMN_NAME 列名,  
COLUMN_TYPE 数据类型,  
DATA_TYPE 字段类型,  
CHARACTER_MAXIMUM_LENGTH 长度,  
IS_NULLABLE 是否为空,  
COLUMN_DEFAULT 默认值,  
COLUMN_COMMENT 备注   
FROM INFORMATION_SCHEMA.COLUMNS 
where table_name = "''' + tableName + '''" ''')

# 4.遍历结果，获取查询的结果
newtxt = '''
/// <summary>
/// ''' + tableName + '''
/// </summary>
[Table("''' + tableName + '''")]
public class ''' + dataModelName + '''
{
    XXXX
}
'''
innertxt = ''
ResultList = cursor.fetchall()
for i in range(len(ResultList)):
    innertxt = innertxt + '''
    /// <summary>
    /// %s
    /// </summary>
    [Column("%s")]
    [DisplayName("%s")]
    public %s %s { get; set; }
    ''' % (ResultList[i][6], ResultList[i][0], ResultList[i][6],
           TypeChange(ResultList[i][2]), 
           ColumnNameChange(ResultList[i][0], prefix))

    # print(ResultList[i])
# 提交如果需要插入语句的时候使用commit
# conn.commit()
# 关闭
cursor.close()
conn.close()

f = open(dataModelName + '.cs', mode="w", encoding="utf8")
f.write(newtxt.replace("XXXX", innertxt))
f.flush()
f.close()

print("生成完成")
